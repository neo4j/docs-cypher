    \begin{align*}
        \operatorname{cosine}(\vec v,\vec u)
        &\coloneqq\tfrac12\left(1 + \hat v\cdot\hat u\right) \\[\jot]
        &\phantom{:}= \dfrac12\left[1 + \dfrac{\vec v\cdot\vec u}{\lVert\vec v\rVert\,\lVert\vec u\rVert}\right]
        \qquad\in\left[0,1\right]
    \end{align*}
