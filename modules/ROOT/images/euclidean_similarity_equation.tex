    \begin{equation*}
        \operatorname{Euclidean}(\vec v,\vec u)
        \coloneqq\dfrac1{1 + \left\lVert\vec v - \vec u\right\rVert^2}
        \qquad\in\left(0,1\right]
    \end{equation*}
